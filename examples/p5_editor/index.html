editor.p5js.org starter

p5 Editor Starter

A paste-ready starter for the p5.js Web Editor. Add the CDN script in index.html, then paste the sketch into sketch.js.

index.html tab
<script src="https://cdn.jsdelivr.net/npm/p5@2/lib/p5.js"></script>
<script src="https://cdn.jsdelivr.net/gh/seb-prjcts-be/p5.gysin@v0.2.0/p5.gysin.min.js"></script>
<script src="sketch.js"></script>
sketch.js core
seedValue = 19600319;

function setup() {
  createCanvas(720, 720);
  noLoop();
  buildPlot();
}

function draw() {
  background(245);
  plot.draw();
}
